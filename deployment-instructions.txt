db.games.find().forEach(function(game) {
	// Disable specialists technology for all games. This will mean
	// that specialists in these games can be purchased by credits, not tokens.
	game.settings.technology.researchCosts.specialists = 'none';

	// Set default research levels for all players
	game.galaxy.players.forEach(function(player) {
		player.research.specialists = {
			level: 1,
			progress: 0
		};
	});
	
	db.games.save(game);
});
