db.games.find().forEach(function(game) {
	game.galaxy.stars.forEach(function(star) {
		star.shipsActual = star.garrisonActual;
		star.ships = star.garrison;

		delete star.garrisonActual;
		delete star.garrison;
	});

	db.games.save(game);
});

db.gamehistories.find().forEach(function(history) {
	if (history.stars) {
		history.stars.forEach(function(star) {
			star.shipsActual = star.garrisonActual;
			star.ships = star.garrison;

			delete star.garrisonActual;
			delete star.garrison;
		});

		db.gamehistories.save(history);
	}
});
