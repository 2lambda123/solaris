db.games.find({
    'conversations': { $ne: null }
}).forEach(function(game) {
    for (let convo of game.conversations) {
        for (let message of convo.messages) {
            let fromPlayer = game.galaxy.players.find(p => p._id.equals(message.fromPlayerId));

            message.fromPlayerAlias = fromPlayer.alias;
        }
    }

    db.games.save(game);
});
