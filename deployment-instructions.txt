db.gamehistories.find({
    'stars.naturalResources': {$type: 'int'}
}).forEach(function(history) {
    for (let star of history.stars) {
        star.naturalResources.economy = star.naturalResources;
        star.naturalResources.industry = star.naturalResources;
        star.naturalResources.science = star.naturalResources;
        
        delete star.naturalResources;
    }

    db.gamehistories.save(history);
});
