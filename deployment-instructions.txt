db.games.find().forEach(function(game) {
	// Disable specialists technology for all games. This will mean
	// that specialists in these games can be purchased by credits, not tokens.
	game.settings.technology.researchCosts.specialists = 'none';
	game.settings.specialGalaxy.specialistsCurrency = 'credits';

	if (game.settings.specialGalaxy.specialistCost === 'cheap') {
		game.settings.specialGalaxy.specialistCost = 'standard';
	}

	// Set default research levels for all players
	game.galaxy.players.forEach(function(player) {
		player.creditsSpecialists = 0; // Set default credit amount.

		player.research.specialists = {
			level: 1,
			progress: 0
		};
	});
	
	db.games.save(game);
});

// Set default value for any time machine history.
db.gamehistories.updateMany({
	'players.credits': {
		$exists: true
	}
}, {
	$set: {
		'players.$[].creditsSpecialists': 0
	}
});
